# Data model (Prisma + domain) implementation plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the PostgreSQL schema in Prisma with an initial migration, add `src/domain` types/DTOs and meal mapping helpers, split unit vs integration tests, and keep `docs/data-model.md` aligned with the schema.

**Architecture:** Prisma models mirror the spec’s relational design (`Household`, `User`, `HouseholdMembership`, `Meal`, `Ingredient`, junctions, `DayPlan`). Domain code uses branded ID types and plain TypeScript interfaces for DTOs; mappers convert persistence-shaped inputs (plain objects typed as inputs, not `Prisma.Meal` in the domain layer if that would couple tests to generated client—use explicit input types in `src/domain/mappers/`). Calendar dates for day plans use **naive calendar dates**: accept `YYYY-MM-DD` strings and persist as `@db.Date` with no timezone conversion in MVP; document that in `docs/data-model.md`.

**Tech Stack:** Bun, TypeScript (strict), Prisma 7 + PostgreSQL, `@prisma/adapter-pg`, existing `src/lib/prisma.ts`.

**Spec:** `docs/superpowers/specs/2026-03-28-data-model-design.md`

```prisma
generator client {

datasource db {

enum IngredientStorageType {
  PANTRY
  REFRIGERATED
  FROZEN
  FRESH

model Household {
  id        String   @id @default(uuid())
  name      String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  memberships   HouseholdMembership[]
  meals         Meal[]
  ingredients   Ingredient[]
  dayPlans      DayPlan[]
}

model User {
  id          String   @id @default(uuid())
  displayName String?
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  memberships HouseholdMembership[]
  mealsCooked MealCookedBy[]
}

model HouseholdMembership {
  userId      String
  householdId String
  user        User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  household   Household @relation(fields: [householdId], references: [id], onDelete: Cascade)

  @@id([userId, householdId])
  @@index([householdId])
}

model Meal {
  id             String   @id @default(uuid())
  householdId    String
  name           String
  description    String   @default("")
  recipeUrl      String?
  imageId        String?
  makesLeftovers Boolean  @default(false)
  isGreasy       Boolean  @default(false)
  isCreamy       Boolean  @default(false)
  isAcidic       Boolean  @default(false)
  createdAt      DateTime @default(now())
  updatedAt      DateTime @updatedAt

  household Household @relation(fields: [householdId], references: [id], onDelete: Cascade)

  heroIngredients MealHeroIngredient[]
  cookedBy        MealCookedBy[]
  dayPlansLunch   DayPlan[]            @relation("DayPlanLunchMeal")
  dayPlansDinner  DayPlan[]            @relation("DayPlanDinnerMeal")

  @@index([householdId])
}

model Ingredient {
  id            String                @id @default(uuid())
  householdId   String
  name          String
  storageType   IngredientStorageType
  perishable    Boolean               @default(false)
  createdAt     DateTime              @default(now())
  updatedAt     DateTime              @updatedAt

  household Household            @relation(fields: [householdId], references: [id], onDelete: Cascade)
  mealHeros MealHeroIngredient[]

  @@unique([householdId, name])
  @@index([householdId])
}

model MealHeroIngredient {
  mealId       String
  ingredientId String
  sortOrder    Int    @default(0)

  meal       Meal       @relation(fields: [mealId], references: [id], onDelete: Cascade)
  ingredient Ingredient @relation(fields: [ingredientId], references: [id], onDelete: Restrict)

  @@id([mealId, ingredientId])
  @@index([mealId])
}

model MealCookedBy {
  mealId String
  userId String

  meal Meal @relation(fields: [mealId], references: [id], onDelete: Cascade)
  user User @relation(fields: [userId], references: [id], onDelete: Restrict)

  @@id([mealId, userId])
  @@index([mealId])
}

model DayPlan {
  id           String   @id @default(uuid())
  householdId  String
  date         DateTime @db.Date
  lunchMealId  String?
  dinnerMealId String?
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  household Household @relation(fields: [householdId], references: [id], onDelete: Cascade)
  lunchMeal Meal?     @relation("DayPlanLunchMeal", fields: [lunchMealId], references: [id], onDelete: Restrict)
  dinnerMeal Meal?    @relation("DayPlanDinnerMeal", fields: [dinnerMealId], references: [id], onDelete: Restrict)

  @@unique([householdId, date])
  @@index([householdId])
}
```

- [ ] **Step 3: Implement** — validate `YYYY-MM-DD` with a regex; build `Date` via **`Date.UTC(y, m - 1, d, 12, 0, 0, 0)`** (noon UTC avoids boundary quirks when drivers truncate to `date`).

```typescript
export function planDateFromYmd(ymd: string): Date {
  const m = /^(\d{4})-(\d{2})-(\d{2})$/.exec(ymd);
  if (!m) throw new RangeError(`Invalid plan date (expected YYYY-MM-DD): ${ymd}`);
  const y = Number(m[1]);
  const mo = Number(m[2]);
  const d = Number(m[3]);
  return new Date(Date.UTC(y, mo - 1, d, 12, 0, 0, 0));
}
```

```typescript
import { describe, expect, test } from 'bun:test';

import { IngredientStorageType } from '@prisma/client';

describe('Prisma schema (integration)', () => {
  test('creates household, user, membership, meal, ingredient, day plan', async () => {
    const suffix = crypto.randomUUID();
    const household = await prisma.household.create({
      data: { name: `H-${suffix}` },
    });
    const user = await prisma.user.create({
      data: { displayName: 'Chef' },
    });
    await prisma.householdMembership.create({
      data: { userId: user.id, householdId: household.id },
    });
    const meal = await prisma.meal.create({
      data: {
        householdId: household.id,
        name: 'Soup',
        description: '',
      },
    });
    const ing = await prisma.ingredient.create({
      data: {
        householdId: household.id,
        name: `tofu-${suffix}`,
        storageType: IngredientStorageType.REFRIGERATED,
        perishable: true,
      },
    });
    await prisma.mealHeroIngredient.create({
      data: { mealId: meal.id, ingredientId: ing.id, sortOrder: 0 },
    });
    await prisma.mealCookedBy.create({
      data: { mealId: meal.id, userId: user.id },
    });
    const plan = await prisma.dayPlan.create({
      data: {
        householdId: household.id,
        date: planDateFromYmd('2026-03-15'),
        lunchMealId: meal.id,
        dinnerMealId: null,
      },
    });
    expect(plan.householdId).toBe(household.id);

    await prisma.$transaction([
      prisma.dayPlan.delete({ where: { id: plan.id } }),
      prisma.mealHeroIngredient.delete({
        where: { mealId_ingredientId: { mealId: meal.id, ingredientId: ing.id } },
      }),
      prisma.mealCookedBy.delete({
        where: { mealId_userId: { mealId: meal.id, userId: user.id } },
      }),
      prisma.meal.delete({ where: { id: meal.id } }),
      prisma.ingredient.delete({ where: { id: ing.id } }),
      prisma.householdMembership.delete({
        where: { userId_householdId: { userId: user.id, householdId: household.id } },
      }),
      prisma.user.delete({ where: { id: user.id } }),
      prisma.household.delete({ where: { id: household.id } }),
    ]);
  });
});
```
